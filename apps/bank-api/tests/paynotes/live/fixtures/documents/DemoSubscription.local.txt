          const readText = (path, fallback) => {
            const node = document.canonical(path);
            if (node && typeof node.value === 'string') {
              return node.value.trim();
            }
            return fallback;
          };
          const readNumber = (path, fallback) => {
            const node = document.canonical(path);
            if (node && typeof node.value === 'number' && Number.isFinite(node.value)) {
              return Math.trunc(node.value);
            }
            return fallback;
          };

          const mandateId = readText('/subscription/paymentMandateDocumentId', '');
          const mandateStatus = readText('/subscription/paymentMandateStatus', '');
          if (!mandateId || mandateStatus !== 'active') {
            return { events: [] };
          }

          const completedCycles = readNumber('/subscription/completedCycles', 0);
          const totalCycles = readNumber('/subscription/totalCycles', 12);
          if (completedCycles >= totalCycles) {
            return { events: [] };
          }

          const req =
            event &&
            event.message &&
            event.message.request &&
            typeof event.message.request === 'object'
              ? event.message.request
              : {};
          const monthlyAmountMinor = readNumber('/subscription/monthlyAmountMinor', 1200);
          const amountMinorInput =
            req.amountMinor != null ? Number(req.amountMinor) : monthlyAmountMinor;
          const amountMinor = Number.isFinite(amountMinorInput)
            ? Math.trunc(amountMinorInput)
            : 0;
          if (!(amountMinor > 0)) {
            return { events: [] };
          }

          const cycleNo = completedCycles + 1;
          const requestId =
            req.requestId != null
              ? String(req.requestId)
              : 'subscription-cycle-' + cycleNo;

            events: [
              {
                type: 'PayNote/Linked Card Charge and Capture Immediately Requested',

payNoteInitialStateDescription:
  summary: |
    ## Demo Subscription (12 months)
    Root card transaction represents the first billing period; mandate enables subsequent monthly charges.
  initialMessage: |
    Thank you for signing up. Get 1 extra free month if you subscribe now.
  details: |
    #### Exact flow
    1. On init, contract emits `PayNote/Capture Funds Requested` to capture the existing attached hold for cycle 1.
    2. In the same init flow, contract emits `Conversation/Document Bootstrap Requested` for `PayNote/Payment Mandate` (cycles 2..N).
    3. Bank asks customer for approval, then emits bootstrap response events.
    4. On `Conversation/Document Bootstrap Completed`, contract stores `paymentMandateDocumentId` and sets status `active`.
    5. Merchant runs `triggerScheduledPayment` once per month (manual simulation) for subsequent periods.
    6. Contract emits `PayNote/Linked Card Charge and Capture Immediately Requested` (local direction: `payerChannel -> payeeChannel`) with stored Payment Mandate id.
    7. `completedCycles` increments only for `PayNote/Card Charge Completed` with `status: succeeded`.
